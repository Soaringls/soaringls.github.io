ffmpeg -i test.mp4 out.gif 
#func2---设置参数
# ss 2 to 12 表示从从视频的第2秒开始, 长度为12秒
# -s         表示用于设定分辨率
# -r         用于设定帧数，通常Gif有15帧左右

#加速:将原始视频4倍播放保存
ffmpeg -i nn.mp4 -filter:v "setpts=0.25*PTS" nn.acc.mp4
#慢速:将原始视频1/4倍播放保存
ffmpeg -i nn.mp4 -filter:v "setpts=4*PTS" nn.slow.mp4
